# 리스크 관리 퀴즈

리스크 관리의 핵심 지표와 포트폴리오 리스크 관리 전략에 대한 이해도를 점검합니다. VaR, Sharpe Ratio, 켈리 기준 등 실무에서 활용되는 개념을 얼마나 정확히 이해하고 있는지 확인하십시오.

## 리스크 측정 지표

<MultipleChoice
  question="VaR 95% 신뢰수준에서 1일 VaR가 1억원이라는 의미는?"
  options={["95% 확률로 손실이 1억원 이내", "5% 확률로 1억원 이상 손실 가능", "평균 손실이 1억원", "최대 손실이 1억원"]}
  answer={1}
  explanation="VaR(Value at Risk)는 특정 신뢰수준에서 일정 기간 동안 발생할 수 있는 최대 손실을 의미합니다. 95% 신뢰수준의 1일 VaR가 1억원이면, 하루 동안 5% 확률로 1억원 이상의 손실이 발생할 수 있다는 뜻입니다."
/>

<TrueFalse
  statement="CVaR(조건부 VaR)은 VaR를 초과하는 손실의 기대값이다."
  answer={true}
  explanation="CVaR(Conditional VaR, Expected Shortfall)은 VaR를 초과하는 영역의 평균 손실을 나타냅니다. 꼬리 리스크(tail risk)를 VaR보다 더 정확하게 포착하며, 서브어디티비티(subadditivity) 조건을 충족하는 일관된 리스크 측도입니다."
/>

<MultipleChoice
  question="Maximum Drawdown이 -40%일 때 원금 회복에 필요한 수익률은?"
  options={["40%", "50%", "약 67%", "100%"]}
  answer={2}
  explanation="원금이 40% 하락하면 잔액은 원금의 60%입니다. 원금 회복을 위해 0.6 × (1 + x) = 1이므로, x = 1/0.6 - 1 = 0.667, 즉 약 66.7%의 수익률이 필요합니다. 손실이 클수록 회복에 필요한 수익률이 기하급수적으로 커집니다."
/>

<CalculationExercise
  question="포트폴리오 기대수익률 12%, 표준편차 18%, 무위험수익률 3%일 때 Sharpe Ratio는?"
  answer={0.5}
  tolerance={0.01}
  unit=""
  hint="Sharpe Ratio = (기대수익률 - 무위험수익률) / 표준편차"
  explanation="Sharpe Ratio = (12% - 3%) / 18% = 9% / 18% = 0.5입니다. Sharpe Ratio는 총변동성 1단위당 초과수익을 나타내며, 값이 높을수록 위험 대비 수익이 우수합니다."
/>

<MultipleChoice
  question="Sortino Ratio와 Sharpe Ratio의 핵심 차이점은?"
  options={["벤치마크 사용 여부", "하방 변동성만 사용", "로그수익률 사용", "연율화 방법"]}
  answer={1}
  explanation="Sortino Ratio는 분모에 총 표준편차가 아닌 하방 표준편차(downside deviation)만 사용합니다. 투자자들이 실제로 우려하는 것은 하락 리스크이므로, Sortino Ratio가 더 실질적인 위험조정 수익률 지표로 평가됩니다."
/>

## 포트폴리오 리스크 관리

<QuizContainer title="포트폴리오 리스크 관리" totalQuestions={5}>

<MultipleChoice
  question="켈리 기준(Kelly Criterion)에서 승률 60%, 보상비율 1.5일 때 최적 베팅 비율은?"
  options={["약 33%", "50%", "60%", "15%"]}
  answer={0}
  explanation="Kelly Fraction f* = p - q/b = 0.6 - 0.4/1.5 = 0.6 - 0.267 = 0.333, 즉 약 33%입니다. 켈리 기준은 장기적으로 자본 성장률을 극대화하는 최적 비율이지만, 실무에서는 변동성을 줄이기 위해 Half-Kelly(절반)를 적용하기도 합니다."
/>

<TrueFalse
  statement="상관계수가 0인 두 자산을 결합하면 포트폴리오 리스크를 줄일 수 있다."
  answer={true}
  explanation="상관계수가 0이면 두 자산의 수익률이 독립적으로 움직입니다. 포트폴리오 분산 = w1²σ1² + w2²σ2² + 2w1w2ρσ1σ2에서 ρ=0이면 교차항이 사라져, 개별 자산의 가중평균 분산보다 포트폴리오 분산이 작아집니다."
/>

<MultipleChoice
  question="2008년 금융위기에서 관찰된 자산간 상관관계의 특징은?"
  options={["자산간 상관관계가 급격히 하락", "자산간 상관관계가 1에 수렴", "변화 없음", "음의 상관관계 강화"]}
  answer={1}
  explanation="위기 시에는 '상관관계 수렴(correlation convergence)' 현상이 발생합니다. 평소에는 낮은 상관관계를 보이던 자산들이 패닉 시점에 동시에 하락하면서 상관계수가 1에 가까워집니다. 이는 분산투자의 효과가 가장 필요한 시점에 그 효과가 약화된다는 것을 의미합니다."
/>

<CalculationExercise
  question="A자산 비중 60%(기대수익률 10%, 표준편차 20%), B자산 비중 40%(기대수익률 5%, 표준편차 10%), 상관계수 0일 때 포트폴리오 기대수익률은?"
  answer={8}
  tolerance={0.01}
  unit="%"
  hint="포트폴리오 기대수익률 = Σ(비중 × 개별 기대수익률)"
  explanation="포트폴리오 기대수익률 = 0.6 × 10% + 0.4 × 5% = 6% + 2% = 8%입니다. 기대수익률은 상관계수와 무관하게 비중의 가중평균으로 결정됩니다."
/>

<TrueFalse
  statement="Trailing Stop은 가격이 상승할 때 손절 기준점도 함께 상승한다."
  answer={true}
  explanation="Trailing Stop은 가격이 상승하면 손절 기준점(stop price)도 함께 올라가지만, 가격이 하락해도 손절 기준점은 내려가지 않는 방식입니다. 이를 통해 수익을 보호하면서도 추가 상승 기회를 놓치지 않을 수 있습니다."
/>

</QuizContainer>
